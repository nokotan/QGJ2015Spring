使用例\+: 
\begin{DoxyCode}
\textcolor{comment}{// int[120][30] の２次元配列を作成し、-1で埋める}

\textcolor{comment}{// [] を使って配列の内容を更新する。}
MyMap[30][20] = 120;
\textcolor{comment}{// [] を使って配列の内容を取得する。}
\textcolor{keywordtype}{int} val = MyMap[20][20];

printf(\textcolor{stringliteral}{"MyMap[20][20] = %d\(\backslash\)n"}, val);

\textcolor{comment}{// MyMap[20][20] = -1}
