template<class Type , size_t Width, size_t Height>
Map<Type> CreateMap ( Type(&)  map[Width][Height])

既存の２次元配列から、Map オブジェクトを作成します。渡された２次元配列はこのタイルマップの間で共有され、 一方を変更するともう一方も変更されます。

使用例:

1 int MapTiles[120][30];
2 MapTiles[20][20] = -1;
3 
4 auto&amp;&amp; MyMap = CreateMap(MapTiles);
5 // これを書くのと同じ
6 // Map<int>&amp;&amp; MyMap { MapTiles };
7 
8 // [] を使って配列の内容を更新する。
9 MyMap[30][20] = 120;
10 // [] を使って配列の内容を取得する。
11 int val = MyMap[20][20];
12 
13 printf("MyMap[20][20] = %d\n", val);
14 // 出力:
15 // MyMap[20][20] = -1
template<class Type , size_t Width, size_t Height>
TemporaryMap<Type, Width, Height> CreateTemporaryMap ( Type(&)  map[Width][Height])

既存の２次元配列から、TemporaryMap オブジェクトを作成します。渡された２次元配列はこのタイルマップの間で共有され、 一方を変更するともう一方も変更されます。

使用例:

1 int MapTiles[120][30];
2 MapTiles[20][20] = -1;
3 
4 auto&& MyMap = CreateTemporaryMap(MapTiles);
5 // これを書くのと同じ
6 // TemporaryMap<int, 120, 30>&amp;&amp; MyMap { MapTiles };
7 
8 // [] を使って配列の内容を更新する。
9 MyMap[30][20] = 120;
10 // [] を使って配列の内容を取得する。
11 int val = MyMap[20][20];
12 
13 printf("MyMap[20][20] = %d\n", val);
14 // 出力:
15 // MyMap[20][20] = -1
